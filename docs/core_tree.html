    func tokenize(s string) []string {
	s = strings.Replace(s, " ", "", -1)
	r := strings.NewReplacer("(", " ( ", ")", " ) ", ":", " : ", ";", " ; ", ",", " , ")
	return strings.Fields(r.Replace(s))
}

    Make a tree from a string

    func Make(s string) (*Tree, error) {
	token := tokenize(s)
	if len(token) == 0 {
		return nil, InvalidTreeError{s}
	}
	t := new(Tree)
	t.Node = newNode()

	var flag bool
	var n, c *Node
	n = t.Node

	for _, v := range token {
		switch v {
		case "(":
			c = newNode()
			n.AddChild(c)
			n = c
			flag = false
		case ")":
			n = n.Father
			flag = false
		case ",":
			c = newNode()
			n.Father.AddChild(c)
			n = c
			flag = false
		case ":":
			flag = true
		case ";":
			break
		default:
			if flag {
				length, err := strconv.ParseFloat(v, 64)
				if err != nil {
					return nil, InvalidTreeError{v}
				}
				n.Length = length
			} else {
				n.Name = v
				flag = false
			}
		}
	}

	if t.Node != n {
		return nil, InvalidTreeError{s}
	}

    Update node.Level and node.Id for tree t

    func (t *Tree) UpdateInfo() {
	size := len(t.Nodes)
	t.Node.Level = 0
	t.Node.Id = size - 1
	for i := size - 2; i >= 0; i-- {
		n := t.Nodes[i]
		n.Level = n.Father.Level + 1
		n.Id = i
	}
}

    func (t *Tree) IsBinary() bool {
	for i := range t.Nodes {
		n := t.Nodes[i]
		if n.IsInternal() && !n.IsBinary() {
			return false
		}
	}
	return true
}

	taxon = make(Taxonmap, len(t.Nodes))

		if n.IsLeaf() {
