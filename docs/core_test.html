	gt, err := Make(gs)
	if err != nil {
		fmt.Println(err)
	}
	st, err := Make(ss)
	if err != nil {
		fmt.Println(err)
	}
	sst, _ := st.SpeciesTree()
	rf, err := RF(gt, sst)

		fmt.Println(err)
	}

		t.Fatal("Incorrect RF")
	}
}
