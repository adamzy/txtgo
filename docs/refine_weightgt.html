    func getX(c *f, wdup, wloss float64) (a, b int) {
	var i int
	s := c.slop
	for i = 0; i < len(s); i++ {
		if s[i] > -wdup {
			a = i
			break
		}
	}
	for ; i < len(s); i++ {
		if s[i] >= wloss {
			b = i
			break
		}
	}
	return
}
